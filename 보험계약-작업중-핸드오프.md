# 보험 계약(가입보험) 기능 — 작업중 핸드오프

작성: 2026-07-07 (회사 PC) · 이어서: 집 노트북
브랜치: `wip/insurance-contracts` (main 미배포 — 프로덕션 영향 없음)

## 이 작업이 하는 것 (사용자 승인 완료 스펙)
고객별 **가입보험 계약**을 스키마로 정규화해서 저장·표기하는 기능.
- 입력: ① **직접 입력(기본)** ② 붙여넣기(토스 ERP '체결 보험' 세로표 / 「날짜 계약자(피보험자) 보험사 상품 (월납 금액)」 한 줄형) ③ 파일(xlsx/csv 자동, PDF·이미지는 붙여넣기 안내)
- 저장 위치: `CRM[계약자명].contracts[]` (E2EE 동기화·백업에 자동 포함)
- 중복은 매칭키(증권번호 우선, 없으면 보험사+상품정규화+계약자+계약일)로 자동 통합
- 계약상태: 정상/실효/해지/철회/만기/삭제 (배지). **타 설계사(모집자≠고현승) 계약은 자동 '삭제' 상태**
- 고객목록: `🛡 보험 N건 · 월 XX만` 노출 + '숨김' 필터
- 고객상세: '가입 보험 계약' 카드(유효/해지·실효/숨김 3구획, 계약별 편집·상태변경)
- 일정: 계약일 기반 가입90일·1주년·2주년 + (본인 피보험) 상령일 자동 이벤트
- 고객 소프트삭제(`cr.dormant`): 연락거부·해지 고객 목록 숨김 + 연락 스케줄 제외

## 결정사항(사용자 동의)
1. 계약 귀속 = **계약자** 고객, 피보험자 다르면 별도표기 (예: `피 김현빈`)
2. 삭제 = **소프트삭제**(상태 플래그, 동기화 tombstone 충돌 회피)
3. 팝업 UX = 한투자산 업로드와 동형 모달, **기본 탭 = 직접 입력**

```js
/* 계약일 기반 도출 이벤트: 가입90일·1주년·2주년 + (본인 피보험)상령일 */
function birthMD(b){const m=String(b||'').match(/(\d{4})-(\d{2})-(\d{2})/);return m?{y:+m[1],mo:+m[2],d:+m[3]}:null;}
function buildContractDerived(){
  const out={signup90:[],signup1y:[],signup2y:[],agechange:[]};
  const y=gcDate.getFullYear(),lo=(y-1)+'-01-01',hi=(y+2)+'-01-01';
  const inR=k=>k>=lo&&k<hi;
  const addD=(iso,d)=>{const t=new Date(iso+'T00:00:00');t.setDate(t.getDate()+d);return dkey(t);};
  const addY=(iso,yy)=>{const t=new Date(iso+'T00:00:00');t.setFullYear(t.getFullYear()+yy);return dkey(t);};
  const custs=Object.values(curCustomers());
  custs.forEach(c=>{
    if(c.crm&&c.crm.dormant)return;
    ((c.crm&&c.crm.contracts)||[]).filter(insActive).forEach(ct=>{
      if(!/^\d{4}-\d{2}-\d{2}$/.test(ct.contractDate||''))return;
      const nm=ct.contractor||c.name,sp=shortProd(ct);
      const push=(cat,iso,lab)=>{if(inR(iso))out[cat].push({id:cat+'::'+ct.id,summary:lab,description:esc((ct.insurer||'')+' '+(ct.product||'')),status:'confirmed',start:{date:iso},end:{date:nextDay(iso)},_cat:cat,_derived:true,_name:nm});};
      push('signup90',addD(ct.contractDate,90),'📌 '+nm+' 가입 90일 · '+sp);
      push('signup1y',addY(ct.contractDate,1),'🎉 '+nm+' 가입 1주년 · '+sp);
      push('signup2y',addY(ct.contractDate,2),'🎉 '+nm+' 가입 2주년 · '+sp);
    });
  });
  const seen=new Set();
  custs.forEach(c=>{
    if(c.crm&&c.crm.dormant)return;
    const cts=((c.crm&&c.crm.contracts)||[]).filter(insActive);
    if(!cts.some(ct=>(ct.insured||ct.contractor||c.name)===c.name))return;   // 본인 피보험 계약이 있을 때만
    const b=birthMD(c.birth);if(!b)return;
    [y-1,y,y+1].forEach(yy=>{const t=new Date(yy,b.mo-1,b.d);t.setMonth(t.getMonth()+6);const iso=dkey(t);   // 상령일=생일+6개월
      if(!inR(iso))return;const key='age::'+c.name+'::'+iso;if(seen.has(key))return;seen.add(key);
      out.agechange.push({id:key,summary:'🎂 '+c.name+' 상령일(보험나이 +1)',status:'confirmed',start:{date:iso},end:{date:nextDay(iso)},_cat:'agechange',_derived:true,_name:c.name});});
  });
  return out;
}
```
